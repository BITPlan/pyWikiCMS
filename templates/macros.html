{# Useful Jinja Template Macros e.g. table4DictList https://stackoverflow.com/a/42848018/1497139 #} {# show html header #} {% macro header(title) -%} {% if title is not none %}{{ title }}{%endif %} {%- endmacro %} {# show html footer #} {% macro footer() -%} {%- endmacro %} {# table for dict list #} {% macro table4DictList(dictList) -%} {% if dictList %} {% for key in dictList[0] %} {% endfor %} {% endif %} {% for dict_item in dictList %} {% for value in dict_item.values() %} {% endfor %} {% endfor %}
{{ key }}
{{ value }}
{%- endmacro %}
